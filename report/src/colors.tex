%Primary color definition
\definecolor{primary_red}{cmyk}{0.1,1.0,0.8,0.0}
\definecolor{primary_black}{cmyk}{0.0,0.0,0.0,1.0}
\definecolor{primary_white}{cmyk}{0.0,0.0,0.0,0.0}

%Secondary color definition in own name notation
\definecolor{orange_light}{cmyk}{0.0,0.25,0.90,0.0}
\definecolor{orange_medium}{cmyk}{0.0,0.58,0.93,0.0}
\definecolor{orange_dark}{cmyk}{0.0,1.00,0.6,0.6}

\definecolor{green_light}{cmyk}{0.4,0.0,1.0,0.0}
\definecolor{green_medium}{cmyk}{0.45,0.0,1.0,0.4}
\definecolor{green_dark}{cmyk}{0.94,0.0,0.59,0.64}

\definecolor{blue_light}{cmyk}{0.7,0.07,0.1,0.0}
\definecolor{blue_medium}{cmyk}{1.0,0.0,0.06,0.4}
\definecolor{blue_dark}{cmyk}{1.0,0.2,0.12,0.73}

\definecolor{violett_light}{cmyk}{0.54,0.93,0.0,0.0}
\definecolor{violett_medium}{cmyk}{0.5,1.0,0.0,0.5}
\definecolor{violett_dark}{cmyk}{0.5,1.0,0.35,0.60}

%Selection of the currently selected color scheme to create secondary_ colors
\DeclareOptionX{colorscheme}[nocolorscheme]{
	\ifthenelse{\equal{#1}{orange}}
	{
		\definecolor{secondary_light}{named}{orange_light}
		\definecolor{secondary_medium}{named}{orange_medium}
		\definecolor{secondary_dark}{named}{orange_dark}
	}
	{
		\ifthenelse{\equal{#1}{green}}
		{
			\definecolor{secondary_light}{named}{green_light}
			\definecolor{secondary_medium}{named}{green_medium}
			\definecolor{secondary_dark}{named}{green_dark}
		}
		{
			\ifthenelse{\equal{#1}{blue}}
			{
				\definecolor{secondary_light}{named}{blue_light}
				\definecolor{secondary_medium}{named}{blue_medium}
				\definecolor{secondary_dark}{named}{blue_dark}
			}
			{
				\ifthenelse{\equal{#1}{violett}}
				{
					\definecolor{secondary_light}{named}{violett_light}
					\definecolor{secondary_medium}{named}{violett_medium}
					\definecolor{secondary_dark}{named}{violett_dark}
				}
				{
					\ifthenelse{\equal{#1}{nocolorscheme}}
					{
						\PackageWarning{tubs*}{Package option colorscheme is not defined, please use either orange, green, blue or violett}
					}
					{
						\PackageWarning{tubs*}{Illegal selection for package option colorscheme: #1, please use either orange, green, blue or violett}
					}
				}
			}
		}
	}
}
