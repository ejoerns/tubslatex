\newcommand{\makebackpage}{}

\DeclareOptionX{TUBStitlepage}[classic]{
	\ifthenelse{\equal{#1}{classic}}
	{
		\renewcommand{\maketitle}
		{
			\begin{TUBStitlepage}
				\TUBSGaussGridTextBox{6}{8}{0}{6}{}{primary_black}{secondary_light!20}{top}
				\TUBSGaussGridTextBox{6}{4}{0}{6}{\sffamily\bfseries\Huge \TUBStitle}{primary_black}{secondary_light!20}{top}
				\TUBSGaussGridTextBox{6}{4}{0}{6}{\sffamily\Huge \TUBSsubject \\ \Large \TUBSdate}{primary_black}{transparent}{bottom}
				\TUBSGaussGridTextBox{0}{4}{0}{6}{\sffamily\huge \TUBSauthor}{primary_black}{secondary_medium!20}{top}
			\end{TUBStitlepage}
		}
		\renewcommand{\makebackpage}
		{
			\begin{TUBSbackpage}
				\TUBSGaussGridTextBox{0}{4}{3}{6}{\TUBSauthor \\ \textbf{\TUBSinstitute} \\ \TUBSaddress}{primary_black}{secondary_medium!20}{bottom}
				\TUBSGaussGridTextBox{0}{4}{0}{3}{}{primary_black}{secondary_medium!20}{bottom}
			\end{TUBSbackpage}
		}
	}
	{
		\ifthenelse{\equal{#1}{picture}}
		{
			\renewcommand{\maketitle}
			{
				\begin{TUBStitlepage}
					\TUBSGaussGridImageBox{6}{8}{0}{6}{./titlepicture}
					\TUBSGaussGridTextBox{6}{4}{0}{6}{\sffamily\bfseries\Huge \TUBStitle}{primary_white}{secondary_dark}{top}
					\TUBSGaussGridTextBox{6}{4}{0}{6}{\sffamily\Huge \TUBSsubject \\ \Large \TUBSdate}{primary_white}{transparent}{bottom}
					\TUBSGaussGridTextBox{3}{4}{0}{6}{\sffamily\huge \TUBSauthor}{primary_white}{secondary_medium}{center}
					\TUBSGaussGridTextBox{0}{3}{0}{6}{}{secondary_dark}{secondary_light}{top}
				\end{TUBStitlepage}
			}
			\renewcommand{\makebackpage}
			{
				\begin{TUBSbackpage}
					\TUBSGaussGridTextBox{0}{3}{3}{6}{\TUBSauthor \\ \textbf{\TUBSinstitute} \\ \TUBSaddress}{primary_white}{secondary_dark}{top}
					\TUBSGaussGridTextBox{3}{4}{0}{6}{}{primary_white}{secondary_medium}{center}
					\TUBSGaussGridTextBox{0}{3}{0}{3}{}{primary_white}{secondary_dark}{bottom}
				\end{TUBSbackpage}
			}
		}
		{
			\ifthenelse{\equal{#1}{alternate}}
			{
				\renewcommand{\maketitle}
				{
					\begin{TUBStitlepage}
						\TUBSGaussGridImageBox{6}{8}{0}{6}{./titlepicture}
						\TUBSGaussGridTextBox{6}{4}{0}{6}{\sffamily\bfseries\Huge \TUBStitle}{primary_black}{secondary_dark!20}{top}
						\TUBSGaussGridTextBox{6}{4}{0}{6}{\sffamily\Huge \TUBSsubject \\ \Large \TUBSdate}{primary_black}{transparent}{bottom}
						\TUBSGaussGridTextBox{3}{4}{0}{6}{\sffamily\huge \TUBSauthor}{primary_black}{secondary_medium!20}{center}
						\TUBSGaussGridTextBox{0}{3}{0}{6}{}{primary_black}{secondary_light!20}{top}
					\end{TUBStitlepage}
				}
				\renewcommand{\makebackpage}
				{
					\begin{TUBSbackpage}
						\TUBSGaussGridTextBox{0}{3}{3}{6}{\TUBSauthor \\ \textbf{\TUBSinstitute} \\ \TUBSaddress}{primary_black}{secondary_dark!20}{top}
						\TUBSGaussGridTextBox{3}{4}{0}{6}{}{primary_black}{secondary_medium!20}{center}
						\TUBSGaussGridTextBox{0}{3}{0}{4}{}{primary_black}{secondary_dark!20}{bottom}
					\end{TUBSbackpage}
				}
			}
			{
				\PackageError{tubs*}{Illegal selection for package option titlepage: #1, possible values are classic, picture, alternate}
			}
		}
	}
}
